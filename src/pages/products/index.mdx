## List all products 	

<Row>

		<Properties> 
			<Property name="query" type="String!">

				`last_3_months`
				`last_year`
			</Property>
			<Property name="after" type="String">
				Returns the elements that come after the specified cursor.
			</Property>
			<Property name="first" type="String">

			</Property>
		</Properties>	

		### Fields 
		<Properties>

				The Product query type.
			</Property>
		</Properties>	

  </Col>
